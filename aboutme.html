import java.util.Scanner;

public class CodeforcesContest188ProblemB{

    public static long calcOddOrEven(long n, long k){
        if (k<=(n+1)/2) {
            return 2*k-1;
        } else {
            return 2*(k-(n+1)/2);
        }

    }
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        long n = scanner.nextLong();
        long k = scanner.nextLong();
        System.out.print(calcOddOrEven(n,k));
        scanner.close();
    }
}
